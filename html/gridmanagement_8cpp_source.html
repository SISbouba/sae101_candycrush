53void initGrid (mat & grid, const size_t & size)
54{
55 srand(time(NULL));
56 grid.resize(size);
57 for (size_t i = 0; i < size; ++i) {
58 grid[i].resize(size);
59 for (size_t j = 0; j < size; ++j) {
60 // Génération aléatoire d'une valeur de bonbon en évitant les alignements initiaux
61 bool estValide;
62 unsigned bonbon;
63
65 do {
66 estValide = true;
67 bonbon = 1 + rand() % (KCandies - 1);
68 // Vérifier l'alignement horizontal (2 à gauche)
69 if (j >= 2 && grid[i][j-1] == bonbon && grid[i][j-2] == bonbon) {
70 estValide = false;
71 }
72 // Vérifier l'alignement vertical (2 au-dessus)
73 if (i >= 2 && grid[i-1][j] == bonbon && grid[i-2][j] == bonbon) {
74 estValide = false;
75 }
76 } while (!estValide);
77
78 grid[i][j] = bonbon;
79 }
80 }
81}

209void gravité(mat& grid) {
210 size_t taille = grid.size();
211
212 for (size_t col = 0; col < taille; col = col + 1) {
213 // Partir du bas et remonter
214 int pos = taille - 1; // Position où on écrit le prochain bonbon
215
216 // Parcourir de bas en haut
217 for (int row = taille - 1; row >= 0; row = row - 1) {
218 if (grid[row][col] != KImpossible) {
219 // Si on a trouvé un bonbon, le placer à la position d'écriture
220 if (row != pos) {
221 grid[pos][col] = grid[row][col];
222 grid[row][col] = KImpossible;
223 }
224 pos = pos - 1;
225 }
226 }
227 }
228}
