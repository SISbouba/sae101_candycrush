38void initGrid (mat & grid, const size_t & size)
39{
40 srand(time(NULL));
41 grid.resize(size);
42 for (size_t i = 0; i < size; ++i) {
43 grid[i].resize(size);
44 for (size_t j = 0; j < size; ++j) {
45 // Génération aléatoire d'une valeur de bonbon en évitant les alignements initiaux
46 bool estValide;
47 unsigned bonbon;
48
49 // Boucle jusqu'à obtenir une valeur valide
50 do {
51 estValide = true;
52 bonbon = 1 + rand() % (KCandies - 1);
53 // Vérifier l'alignement horizontal (2 à gauche)
54 if (j >= 2 && grid[i][j-1] == bonbon && grid[i][j-2] == bonbon) {
55 estValide = false;
56 }
57 // Vérifier l'alignement vertical (2 au-dessus)
58 if (i >= 2 && grid[i-1][j] == bonbon && grid[i-2][j] == bonbon) {
59 estValide = false;
60 }
61 } while (!estValide);
62
63 grid[i][j] = bonbon;
64 }
65 }
66}

158// Applique la gravité sur la grille (Seulement pour le mode Clear)
159void gravité(mat& grid) {
160 size_t taille = grid.size();
161
162 for (size_t col = 0; col < taille; col = col + 1) {
163 // Partir du bas et remonter
164 int pos = taille - 1; // Position où on écrit le prochain bonbon
165
166 // Parcourir de bas en haut
167 for (int row = taille - 1; row >= 0; row = row - 1) {
168 if (grid[row][col] != KImpossible) {
169 // Si on a trouvé un bonbon, le placer à la position d'écriture
170 if (row != pos) {
171 grid[pos][col] = grid[row][col];
172 grid[row][col] = KImpossible;
173 }
174 pos = pos - 1;
175 }
176 }
177 }
178}
